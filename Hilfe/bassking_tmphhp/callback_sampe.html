Beispiel eines Fenster-Callbacks.

Structure _TrackInfo
  File.s
  Title.s
  Artist.s
  Album.s
  Year.s
  Comment.s
  Track.s
  Genre.s
EndStructure
Global TrackInfo._TrackInfo

Procedure WindowCallback(hWnd, Msg, wParam, lParam)
  Protected iResult.l = #PB_ProcessPureBasicEvents
 
  If Msg = #WM_COPYDATA And lParam
    Protected i.i
    Protected *CDS.COPYDATASTRUCT = lParam
   
    If *CDS
      Select *CDS\dwData

        Case #BKR_TagFile
          TrackInfo\File = PeekS(*CDS\lpData)
        Case #BKR_TagTitle
          TrackInfo\Title = PeekS(*CDS\lpData)
        Case #BKR_TagArtist
          TrackInfo\Artist = PeekS(*CDS\lpData)
        Case #BKR_TagAlbum
          TrackInfo\Album = PeekS(*CDS\lpData)
        Case #BKR_TagYear
          TrackInfo\Year = PeekS(*CDS\lpData)
        Case #BKR_TagComment
          TrackInfo\Comment = PeekS(*CDS\lpData)
        Case #BKR_TagTrack
          TrackInfo\Track = PeekS(*CDS\lpData)
        Case #BKR_TagGenre
          TrackInfo\Genre = PeekS(*CDS\lpData)

        Case #BKR_Start
          ; Wiedergabe wurde gestartet
          ; Channel: *CDS\lpData
        Case #BKR_Stop
          ; Wiedergabe wurde beendet
          ; Channel: *CDS\lpData

        Case #BKR_End
          ; BassKing wurde Beendet

        Case #BKR_Volume
          ; Lautstärke wurde verändert
          ; Aktueller Integerwert steht in *CDS\lpData

        Case #BKR_Preferences
          ; Plugin-Einstellungen Öffnen

      EndSelect
    EndIf
   
    iResult = 1
  EndIf
 
  ProcedureReturn iResult
EndProcedure
